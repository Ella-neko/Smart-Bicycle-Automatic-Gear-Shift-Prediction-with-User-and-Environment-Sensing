#include <Servo.h>

Servo myservo; // 建立Servo物件，控制伺服馬達
 int reset = 0;   //復位檢查
 int start = 0;   //復位尋找角度起始點
 int angle = 0;   //轉動到的角度
 int RR = 0;     //現在檔位
 int RR1 = 0;    //前次檔位
 int last_state = 0;  //前次檔位馬達角度
 int back = 0;//退檔紀錄
  
void last_degree(){     //紀錄前次檔位馬達角度(RR1+1 = 實際第幾檔)
  if(RR1==6){
    last_state=145; 
  }
  if(RR1==5){
    last_state=130; 
  }
  if(RR1==4){
    last_state=115; 
  }
  if(RR1==3){
    last_state=105; 
  }
  if(RR1==2){
    last_state=85; 
  }
  if(RR1==1){
    last_state=60; 
  }
  if(RR1==0){
    last_state=0; 
  }
  //Serial.print("last_state = ");
  //Serial.println(last_state);
}
void motorsetup() 
{ 
 // Serial.begin(9600);
  myservo.attach(9); // 連接數位腳位9，伺服馬達的訊號線
   if(start == 0){    //復位
    for(angle = 180; angle >= 0; angle--){
     myservo.write(angle); 
     //Serial.println(angle);
     delay(10);
     if(angle==0){
    reset = 1;
    }
   }
   if(reset==1){
    reset = 1;
    Serial.println("已復位");
   }
 }
  
}

 if(reset==1){

     for(angle = 180; angle >= 105; angle--){
     myservo.write(angle);

     delay(10);
   }
   back = 1;
      }
   if(RR1<=RR){
     if(back==1){
        last_state-=10;
        back = 0;
      }
    for(angle = last_state; angle <= 115; angle++){
     myservo.write(angle);

 }
  
   if(d == 6){
